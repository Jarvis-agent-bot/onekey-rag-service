interface PendingTx {
  id: string
  chainId: number
  tx: { from: string; to: string; data: string; value: string }
  resolve: (value: unknown) => void
  reject: (reason: unknown) => void
}

const pendingTxs = new Map<string, PendingTx>()

function interceptEthereum() {
  const originalProvider = window.ethereum
  if (!originalProvider) return

  const originalRequest = originalProvider.request.bind(originalProvider)

  window.ethereum.request = async (args: { method: string; params?: unknown[] }) => {
    // æ‹¦æˆªå‘é€äº¤æ˜“
    if (args.method === 'eth_sendTransaction' && args.params?.[0]) {
      const tx = args.params[0] as Record<string, string>
      const chainId = await originalRequest({ method: 'eth_chainId' })

      return new Promise((resolve, reject) => {
        const txId = crypto.randomUUID()

        pendingTxs.set(txId, {
          id: txId,
          chainId: parseInt(chainId, 16),
          tx: { from: tx.from || '', to: tx.to || '', data: tx.data || '0x', value: tx.value || '0x0' },
          resolve, reject,
        })

        // é€šçŸ¥æ‰©å±•è¿›è¡Œåˆ†æ
        window.postMessage({ type: 'TX_ANALYZER_INTERCEPT', payload: { txId, chainId: parseInt(chainId, 16), tx } }, '*')
      })
    }

    return originalRequest(args)
  }

  // ç›‘å¬æ‰©å±•å“åº”
  window.addEventListener('message', (event) => {
    if (event.data.type === 'TX_ANALYZER_RESPONSE') {
      const { txId, action } = event.data.payload
      const pending = pendingTxs.get(txId)

      if (pending) {
        if (action === 'approve') {
          originalRequest({ method: 'eth_sendTransaction', params: [pending.tx] })
            .then(pending.resolve).catch(pending.reject)
        } else {
          pending.reject(new Error('Transaction rejected by user'))
        }
        pendingTxs.delete(txId)
      }
    }
  })
}
